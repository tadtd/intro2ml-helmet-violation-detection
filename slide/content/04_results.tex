\vspace{0.4em}

  \midrule
  \multirow{3}{*}{RT-DETR}
    & motorbike  & 0.445 & 0.362 & 0.365 & 0.940 & 0.526 \\

  \item RT-DETR recall cao (0.739) nhưng precision chỉ 0.180 $\Rightarrow$ \textbf{rất nhiều báo nhầm}
\end{itemize}

\vspace{0.2em}
\begin{block}{Kết luận}\small
  Chọn \textbf{YOLO}: tốt nhất ở đúng lớp \textit{non-helmet} (F1 0.611) và nhanh nhất (36.4 FPS).
\end{block}

% ── Slide 4: Phân tích lỗi ────────────────────────────────────────────────────
\begin{frame}{Phân tích lỗi và overfitting}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
  \textbf{Overfitting}
  \begin{itemize}
    \item \textbf{YOLO}: ổn định, hội tụ tốt
    \item \textbf{Faster R-CNN}: overfit rõ sau epoch 2
    \item \textbf{RT-DETR}: dao động, chưa ổn định
  \end{itemize}
  \vspace{0.4em}
  \textbf{Nguyên nhân}
  \begin{itemize}
    \item Mất cân bằng: \textit{helmet} $\gg$ \textit{non-helmet}
    \item Vùng đầu nhỏ, bị che, ảnh xa
  \end{itemize}
\end{column}
\begin{column}{0.47\textwidth}
  \textbf{Lỗi thường gặp}
  \begin{itemize}
    \item \textbf{FN}: bỏ sót vùng đầu nhỏ hoặc bị che
    \item \textbf{FP}: bbox thừa trên vùng giống mũ
    \item \textbf{Nhầm lớp}: helmet $\leftrightarrow$ non-helmet
    \item \textbf{Sai bbox}: chưa sát đối tượng rìa ảnh
  \end{itemize}
  \vspace{0.4em}
  \begin{block}{Tổng kết}
    YOLO $\to$ triển khai\\
    RT-DETR $\to$ cần tối ưu tốc độ\\
    Faster R-CNN $\to$ baseline, FPS thấp
  \end{block}
\end{column}
\end{columns}

\end{frame}
