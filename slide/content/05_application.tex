% ── Slide 2: Các thành phần hệ thống ─────────────────────────────────────────
\begin{frame}{Các thành phần hệ thống}

\begin{columns}[T]
\begin{column}{0.52\textwidth}
  \textbf{Luồng xử lý}
  \begin{itemize}\small
    \item \textbf{Traefik} -- gateway, TLS termination
    \item \textbf{Next.js} -- giao diện web
    \item \textbf{Auth} -- xác thực (Supabase Auth)
    \item \textbf{Ingestion} -- nhận file, lưu metadata
    \item \textbf{Orchestration} -- tạo tác vụ $\to$ Redis
    \item \textbf{Inference Worker} -- chạy mô hình, ghi kết quả
    \item \textbf{Notification} -- cập nhật giao diện
  \end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
  \begin{block}{Dịch vụ ngoài cụm}
    \textbf{Supabase}: DB, Auth, Storage, Realtime
  \end{block}
  \vspace{0.3em}
  \begin{block}{Dịch vụ trong cụm}
    \textbf{Redis}: hàng đợi tác vụ, cache
  \end{block}
  \vspace{0.3em}
  \begin{block}{Thiết kế}
    Tính toán trên \textbf{GKE}; dữ liệu quản lý bởi \textbf{Supabase}
  \end{block}
\end{column}
\end{columns}

\end{frame}

  \begin{enumerate}\small
    \item Push code $\to$ \textbf{GitHub}
    \item \textbf{GitHub Actions} build Docker image
    \item Đẩy lên \textbf{Artifact Registry}
    \item Update workload trên \textbf{GKE}
  \end{enumerate}
