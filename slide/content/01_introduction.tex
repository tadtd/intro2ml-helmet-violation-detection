% ── Slide 1: Bài toán ───────────────────────────────────────────────────────
\begin{frame}{Bài toán và mục tiêu đồ án}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
  \textbf{Vấn đề thực tiễn}
  \begin{itemize}
    \item Vi phạm không đội mũ bảo hiểm là nguyên nhân hàng đầu làm tăng mức độ nghiêm trọng của TNGT tại Việt Nam
    \item Hiện tại: giám sát thủ công bởi CSGT hoặc rà soát camera $\Rightarrow$ tốn nhân lực, dễ bỏ sót
  \end{itemize}
  \vspace{0.5em}
  \textbf{Mục tiêu đồ án}
  \begin{itemize}
    \item Xây dựng bộ dữ liệu ảnh giao thông có gán nhãn
    \item Huấn luyện \textbf{YOLOv8} đạt độ chính xác phù hợp real-time
    \item Tối ưu mô hình qua \textbf{ONNX Runtime}
    \item Triển khai hệ thống microservices trên \textbf{GKE}
  \end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
  \begin{block}{Giải pháp đề xuất}
    Kết hợp camera giám sát sẵn có với mô hình Deep Learning để tự động phát hiện vi phạm, cung cấp minh chứng trực quan hỗ trợ xử phạt.
  \end{block}
\end{column}
\end{columns}

\end{frame}
